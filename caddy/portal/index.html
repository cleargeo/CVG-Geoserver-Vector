Connection Examples
Ready-to-use code for Python, JavaScript, QGIS, cURL, and GDAL/OGR.
# pip install owslib
from owslib.wfs import WebFeatureService
wfs = WebFeatureService("https://vector.cleargeo.tech/geoserver/wfs", version="2.0.0")
print(list(wfs.contents.keys()))
resp = wfs.getfeature(typename=["workspace:layer"], outputFormat="application/json")
import json; data = json.loads(resp.read())
print(data["totalFeatures"], "features")
// OpenLayers WFS example
import VectorLayer from "ol/layer/Vector"; import VectorSource from "ol/source/Vector"; import GeoJSON from "ol/format/GeoJSON";
const src = new VectorSource({url:"https://vector.cleargeo.tech/geoserver/ows?SERVICE=WFS&VERSION=2.0.0&REQUEST=GetFeature&typeName=ws:layer&outputFormat=application/json",format:new GeoJSON()});
const layer = new VectorLayer({source:src});
